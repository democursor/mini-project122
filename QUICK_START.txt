🎯 GOAL: Get your full-stack research platform running in 2 minutes

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 PREREQUISITES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ Python 3.12 installed
✅ Node.js 18+ installed
✅ Virtual environment (vnv) created
✅ Neo4j running (optional)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 STEP 1: START BACKEND (Terminal 1)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Activate virtual environment:
   > vnv\Scripts\activate

2. Start FastAPI server:
   > python run_api.py

✅ Backend running on: http://localhost:8000
✅ API Docs available at: http://localhost:8000/docs

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎨 STEP 2: START FRONTEND (Terminal 2)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Navigate to frontend:
   > cd frontend

2. Install dependencies (FIRST TIME ONLY):
   > npm install

3. Start development server:
   > npm run dev

✅ Frontend running on: http://localhost:3000

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🌐 STEP 3: OPEN BROWSER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Open: http://localhost:3000

You should see:
✅ Dashboard with statistics
✅ Sidebar navigation
✅ Modern, clean interface

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🧪 STEP 4: TEST THE SYSTEM
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Click "Upload" in sidebar
2. Drag & drop a PDF file
3. Click "Upload and Process"
4. Wait for processing to complete
5. Try "Search" to find content
6. Try "AI Assistant" to ask questions

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ Upload          - Drag & drop PDF upload
✅ Documents       - Manage research papers
✅ Search          - Semantic search
✅ Knowledge Graph - Explore relationships

🔧 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

❌ Backend won't start:
   > pip install -r requirements.txt

❌ Frontend won't start:
   > cd frontend
   > rm -rf node_modules package-lock.json
   > npm install

❌ CORS errors:
   → Already configured, just restart backend

❌ Port already in use:
   → Kill the process or change port
